package java3;
import java.util.*;
import java.util.regex.Pattern;
public class GuessData {
	public static boolean isNumeric(String str){
		   for(int i=str.length();--i>=0;){
		      int chr=str.charAt(i);
		      if(chr<48 || chr>57)
		         return false;
		   }
		   return true;
		}
	public static int guess(int x){
		int min=0,max=100,mid;
		int count=0;
		while(min<=max){
			count++;
			mid=(min+max)/2;
			if(x<mid)
				max=mid-1;
			else if(x>mid)
				{
				min=mid+1;
				}
			else
				break;
			
		}	
		return count;
	}
